# Undo the commit but keep your code changes

# Then fix whatever you need and push again

Undo last push and discard changes completely
# Nuclear option - deletes your changes entirely
git reset --hard HEAD~1

git push -f origin main

# Creates a new commit that undoes the last one
# Does NOT rewrite history
git revert HEAD
